  \frametitle{ Tricritical Ising model}
  Hamiltonian of the model
  \begin{equation}
    \label{eq:1}
    H = -\beta \sum_{<i,j>}\sigma_i\sigma_j - \mu \sum_{i}(\sigma_i)^2  
  \end{equation}
  Partition function
  \begin{equation}
    \label{eq:2}
    Z=\sum_{\mathrm{conf}} W[\mathrm{conf}]=\sum_{\mathrm{conf}} e^{-\frac{H[\mathrm{conf}]}{kT}}
  \end{equation}
